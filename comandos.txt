git init = Inicia o repositório.

git status = Mostra a situação dos arquivos no git.

git add (arquivo) = Adiciona arquivos ao git.
    git add . = Adiciona tudo que não foi adicionado ainda.

git commit = "Commita" o arquivo. Envia os arquivos (ou suas alterações) para o repo local.
    git commit -m ("mensagem") = "Commita" o primeiro arquivo com mensagem.
    git commit (arquivo) -m ("mensagem") = "Commita" o arquivo com mensagem.
    git commit -a = "Commita" todos os arquivos.
    git commit -a -m ("mensagem") = "Commita" todos os arquivos com mensagem.
    git commit --amend -m ("nova mensagem") = Altera a mensagem do commit anterior.

git diff = Mostra alterações ainda não adicionadas ao Stage.
    git diff --staged = Mostra alterações que estão no Stage.

git branch = Lista todos os branches.
    git branch -M (nome do branch) = Renomeia o branch atual, forçando a alteração. A primeira e principal costuma ser a main.
    git branch (nome do branch) = Cria um branch novo.
    git branch -d/--delete (nome do branch) = Apaga o branch selecionado.
    git branch -f/--force (nome do branch) (referência) = Move à força o branch selecionado para a referência.

git remote = É um repo remoto.
    git remote -v = Mostra a url remota após o nome.
    git remote add origin (link do github) = Sincroniza o projeto adicionando uma origem a ele.
    git remote rename (antigo) (novo) = Muda o nome do repo remoto.
    git remote set-url (nome) (nova url) = Muda a url do projeto.
    git remote add (nome) (url) = adiciona um novo repo remoto. 
    git remote rm (nome) = Remove um repo remoto.

git push = Envia os arquivos (ou suas alterações) locais para o repo remoto.
    git push -u origin (branch) = Dá um push, na branch selecionada.
    git push -u origin main = Nesse caso, dá um push, na branch principal.
    git push -f origin main = Força o push, sobrescrevendo o que está no GitHub.

git pull = Baixa as alterações do repo remoto. 
    git pull origin main --rebase = Baixa o repo remoto e reaplica os commits locais por cima.

git fetch = Baixa alterações do repo remoto sem aplicá-las no branch atual.
    git fetch (repositório) = Baixa as alterções de um repositório específico sem alterar a branch atual.
    git fetch --all = Baixa as alterções de todos os repositórios sem alterar a branch atual.

git stash = Guarda temporariamente alterações não commitadas.
    git stash pop = Recupera as alterações guardadas.

git log = Mostra o histórico de alterações do repo. "Q" (ou "Ctrl+C" no Linux) sai desse modo.
    git log --oneline -1 = Mostra o commit atual (onde o HEAD está).
    git log --oneline --graph --all = Mostra onde os branches se separam.

git reflog = Mostra o histórico de movimentação do HEAD.

git mv (arquivo antigo) (arquivo/local novo) = Move um arquivo (Ctrl+X). Pode ser usado para renomear o arquivo.

git rm (arquivo) = Remove o arquivo.

git clean = Remove arquivos não rastreados pelo Git.
    git clean -fd = Remove arquivos e pastas não rastreados pelo Git.
    git clean -n = Simula a remoção, exibindo uma lista dos arquivos que seriam excluídos sem removê-los.
    git clean -fx = Remove arquivos não rastreados e também arquivos que estão listados no .gitignore.

git clone (url) = Clona o repo inteiro em uma pasta. Utilizado ao entrar em um projeto em andamento.
    git clone (url) . = Clona o repo dentro do diretório atual (só se a pasta estiver vazia).

git checkout (arquivo) = Restaura o arquivo para o último commit, descartando alterções locais.
git checkout (branch) = Troca para o branch selecionado.
    git checkout -b (branch) = Cria e muda para o branch criado.
git checkout (commit) = Desanexa o HEAD, movendo-o para um commit ao invés de branch.
    git checkout (commit/branch)^ = Desanexa o HEAD, movendo-o para o commit anterior.
    git checkout (commit/branch)^(num) = Desanexa o HEAD, movendo-o para um dos commits pais de um merge, conforme o número informado.
    git checkout (commit/branch)^^ = Desanexa o HEAD, movendo-o para commit anterior do anterior (o avô do commit).
    git checkout (commit/branch)~(num) = Desanexa o HEAD, movendo-o para commit anterior equivalente ao número.

git reset origin/main = Utilizado para voltar ao último commit. Se não especificado, utiliza o "--mixed" automaticamente.
git reset (referência) = Reverte alterações no histórico realizadas após a referência. Se não especificado, utiliza o "--mixed" automaticamente.
    git reset --soft origin/main = Mantém os arquivos e o Stage, mas volta o histórico.
    git reset --mixed origin/main = Mantém os arquivos, mas limpa o Stage e volta o histórico.
    git reset --hard origin/main = Apaga os arquivos, limpa o Stage e volta o histórico.

git revert (referência) = Cria um novo commit que desfaz as alterações da referência, sem apagar o histórico anterior.
git revert HEAD =  Cria um novo commit para desfazer o último commit.
git revert -n/--no-commit (referência) = Reverte as alterações da referência, deixando o commit aberto para alterações.

git switch (branch) = Troca para o branch selecionado. Substitui o "git checkout (branch)"
    git switch -c (novo branch) = Cria e muda para o branch criado.

git restore (arquivo) = Restaura o arquivo para o último commit, descartando alterações locais.
    git restore -s (ID do commit) (arquivo) = Restaura o arquivo para o commit selecionado.

git merge (nome do branch) = Integra o branch atual com as informações do branch selecionado.
Se houver divergências cria um merge commit, se não, faz um fast-foward (apenas move o ponteiro do branch selecionado com o atual).
    git merge --no-ff (nome do branch) = Não faz um fast-foward (mesmo sendo possível). Gera um merge commit.

git rebase (nome do branch) = Reescreve o histórico, copiando o branch atual para o branch selecionado. Criando um histórico linear.
    git rebase -i (referência) = Abre um painel que permite editar, juntar, apagar e reordenar os commits que serão copiados no local selecionado.
        pick: Mantém o commit como está.
        reword: Mantém o commit, mas permite mudar a mensagem do texto.
        s/squash: Junta o commit selecionado com o commit de cima.
        d/drop: Deleta o commit do histórico.

git cherry-pick (referência) = Copia as referências selecionadas de outro branch para o HEAD.

git tag (nome da tag) = Cria uma marcação permanente no commit atual.
    git tag -a (nome da tag) -m ("mensagem") = Cria uma marcação permanente no commit atual, contendo data e informações do autor da tag.
    git tag (nome da tag) (referência) = Cria uma marcação permanente na referência selecionada.

git describe = Procura tags anotadas e escreve com base no commit atual onde está a tag anterior mais próxima.
    git describe (referência) = Procura tags anotadas e escreve com base na referência selecionada onde está a tag anterior mais próxima.

git show = Exibe todo o histórico e as alterações do HEAD.
    git show (referência) = Exibe o histórico e alterações da referência selecionada.
    git show --name-only (nome da tag ou commit) = Mostra o nome do arquivo modificado.
    git show -s (referência) = Exibe apenas a mensagem da alteração.

git blame (arquivo) = Mostra quem alterou cada linha do arquivo.

.gitignore = Arquivo utilizado para ignorar arquivos/pastas no Git.
